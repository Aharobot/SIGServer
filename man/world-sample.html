世界ファイルサンプル


<?xml version="1.0" encoding="utf8"?>
<!-- 名前は"myworld"。 -->
<world name="myworld">

  <!--; 重力加速度を指定-->
  <gravity x="0.0" y="-9.8" z="0.0"/>

  <!-- Robot.xml において定義したエージェントインスタンスを作成-->
  <instanciate class="Robot.xml">
    <!-- set-attr-value で属性値を指定 -->

    <!-- エージェント名はRobot1 -->
    <set-attr-value name="name" value="Robot1"/>
    <!-- 物理シミュレーションを適用するか？ -->
    <set-attr-value name="dynamics" value="true"/>

    <!-- 最初の位置を(3.0, 0.3, 2.2)に設定-->
    <set-attr-value name="x" value="3.0"/>
    <set-attr-value name="y" value="0.3"/>
    <set-attr-value name="z" value="2.2"/>

    <!-- コントローラの指定 -->
    <set-attr-value name="language" value="c++"/>
    <set-attr-value name="implementation" value="../model/samples/JointTest.so"/> 

  </instanciate>

  <!-- Robot.xml において定義したエージェントインスタンスを作成-->
  <instanciate class="Robot.xml">

    <!-- エージェント名はRobot2 -->
    <set-attr-value name="name" value="Robot2"/>
    <set-attr-value name="dynamics" value="true"/>
    <set-attr-value name="language" value="c++"/>
    <set-attr-value name="implementation" value="../model/samples/GoAround.so"/> 

    <!-- 最初の位置を(3.0, 5.0, 2.0)に設定-->
    <set-attr-value name="x" value="3.0"/>
    <set-attr-value name="y" value="5.0"/>
    <set-attr-value name="z" value="2.0"/>
  </instanciate>

  <!-- Desk.xmlにおいて定義した机エンティティインスタンスを作成 -->
  <instanciate class="Desk.xml">

    <set-attr-value name="name" value="Desk1"/>
    
    <!-- 最初の位置を(-10.0, 2.0, 2.0)に設定-->
    <set-attr-value name="x" value="-10.0"/>
    <set-attr-value name="y" value="2.0"/>
    <set-attr-value name="z" value="2.0"/>
  </instanciate>

</world>
