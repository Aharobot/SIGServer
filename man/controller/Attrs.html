15 double AgentController::onAction(ActionEvent &evt)
16 {
17 	try {
18 
19 		SimObj *my = getObj(myname());
20 
21 		double x = my->x();
22 		double y = my->y();
23 		double z = my->z();
24 		LOG_MSG(("position = (%f, %f, %f)", x, y, z));
25 
26 		double qw = my->qw();
27 		double qx = my->qx();
28 		double qy = my->qy();
29 		double qz = my->qz();
30 		LOG_MSG(("quaternion = (%f, %f, %f, %f)", qw, qx, qy, qz));
31 
32 		if (my->dynamics()) {
33 			double fx = my->fx();
34 			double fy = my->fy();
35 			double fz = my->fz();
36 			LOG_MSG(("force = (%f, %f, %f)", fx, fy, fz));
37 
38 			double tqx = my->tqx();
39 			double tqy = my->tqy();
40 			double tqz = my->tqz();
41 			LOG_MSG(("torque = (%f, %f, %f)", tqx, tqy, tqz));
42 		}
43 
44 	} catch(SimObj::NoAttributeException &) {
45 		
46 	} catch(SimObj::AttributeReadOnlyException &) {
47 		
48 	} catch(SimObj::Exception &) {
49 		
50 	}
51 	return 1.0; 
52 }
