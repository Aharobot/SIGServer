19 double Watcher::onAction(ActionEvent &evt)
20 {
21 	static int cnt = 0;
22 	
23 	try {
24 		SimObj *my = getObj(myname());
25 
26 		if (!my || my->dynamics()) { return 5.0; }
27 
28 		typedef std::vector<std::string> C;
29 		C names;
30 		bool detected = detectEntities(names);
31 
32 		if (detected) { 
33 			LOG_MSG(("detected %d", names.size()));
34 			for(C::iterator i=names.begin(); i!=names.end(); i++) {
35 				std::string name = *i;
36 				LOG_MSG(("%s detected", name.c_str()));
37 				sendText(evt.time(), NULL, name.c_str());
38 			}
39 		} else {
40 			LOG_MSG(("not detected"));
41 		}
42 
43 		const double R = 5.0;
44 		double angle = 2*PI*cnt*0.01;
45 		double xx = R*sin(angle);
46 		double yy = my->y();
47 		double zz = R*cos(angle);
48 		
49 		LOG_MSG(("pos (%f, %f, %f)", xx, yy, zz));
50 		my->setPosition(xx, yy, zz);
51 		my->setAxisAndAngle(0.0, 1.0, 0.0, angle+PI/2);
52 
53 		
54 	} catch(SimObj::Exception &e) {
55 		LOG_ERR(("%s : ERROR %s", myname(), e.msg()));
56 	}
57 
58 	cnt++;
59 	return 0.1; 
60 }
