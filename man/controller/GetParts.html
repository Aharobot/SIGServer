23 
24 			Vector3d pos;

26 
27 			LOG_MSG(("parts(%s) : pos(%f, %f, %f)",
28 				 parts->name(),

30 
31 			Rotation r;

33 
34 			const dReal *q = r.q();
35 			LOG_MSG(("parts(%s) : quaternion(%f, %f, %f)",
36 				 parts->name(),
37 				 q[0], q[1], q[2], q[3]));
38 		}
39 		else
40 		{
41 			LOG_MSG(("(%s) not found", partsName));
42 		}
43 
44 	} catch(SimObj::Exception &) {
45 
46 
47 	}
48 	return 1.0; 
49 }
