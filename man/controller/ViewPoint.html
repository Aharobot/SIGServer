 3 #include <sigverse/Controller.h>
 4 #include <sigverse/ControllerEvent.h>
 5 #include <sigverse/SimObj.h>
 6 
 7 class AgentController : public Controller
 8 {
 9 public:
10 	double 	onAction(ActionEvent &evt);
11 };
12 
13 /* #################################################
14  *
15  * 左右の目の視点・視線を得る
16  *
17  * 視点はエージェント座標系、視線は視線座標系。
18  *
19  * ################################################ */
20 
21 double AgentController::onAction(ActionEvent &evt)
22 {
23 	
24 	try {
25 		SimObj *my = getObj(myname());
26 
27 		/* 視点を得る(エージェント座標系)*/
28 		Vector3d rvp;
29 		Vector3d lvp;
30 		my->getRightViewPoint(rvp); // 右目
31 		my->getLeftViewPoint(lvp);  // 左目
32 
33 		// 右目の視線を得る・設定する(視線座標系)
34 		Vector3d rvv;
35 		my->getRightViewVector(rvv);
36 		// 正面
37 		my->setRightViewVector(Vector3d(0, 0, -1));
38 
39 		// 左目の視線を得る・設定する(視線座標系)
40 		Vector3d lvv;
41 		my->getLeftViewVector(lvv);
42 		// 正面
43 		my->setLeftViewVector(Vector3d(0, 0, -1));
44 		
45 	} catch(SimObj::Exception &) {
46 		
47 	}
48 	return 1.0; 
49 }
50 
51 extern "C"  Controller * createController ()
52 {
53 	return new AgentController;
54 }
